
#(define style_table '(
    (hihat cross #f 5)
    (openhihat cross "open" 5)
    (closedhihat cross "stopped" 5)
    (pedalhihat cross #f -5)

    (ridecymbal cross #f 4)
    (ridebell triangle #f 4)
    (crashcymbal xcircle #f 5)

    (snare default #f 1)
    (bassdrum default #f -3)

    (himidtom default #f 3)
    (lowmidtom default #f 2)
    (highfloortom default #f -1)
    (lowfloortom default #f -2)
))
up = \drummode {
    <sn>8 <sn>8 <sn>8 <sn>16 <sn>16 <sn>8 <sn>8 <sn>4
    <sn>16 <sn>16 <sn>16 <sn>16 \tuplet 3/2 { <sn>8 <sn>8 <sn>8 } <sn>8 <sn>8 <sn>4
    \set repeatCountVisibility = #(every-nth-repeat-count-visible 0)
    \repeat percent 1 {
        <sn>16 <sn>16 <sn>16 <sn>16 \tuplet 3/2 { <sn>8 <sn>8 <sn>8 } <sn>8 <sn>8 <sn>4
        \tuplet 3/2 { <sn>4 <sn>8 } <sn>8 <sn>16 <sn>16 <sn>8 <sn>8 <sn>4
    }
    <sn>8 <sn>8 <sn>4 <sn>4 r8 <sn>8
    r8 <sn>8 r16 <sn>8. <sn>8. <sn>16 <sn>8 <sn>8
    \tuplet 3/2 { <sn>8 <sn>8 <sn>8 } <sn>8 <sn>8 <sn>16 <sn>16 <sn>8 <sn>4
    <sn>4 r8 <sn>8 r8. <sn>16 r16 <sn>16 <sn>16 <sn>16
    r16 <sn>8 <sn>16 r4 <sn>16 <sn>16 <sn>8 <sn>4
}
down = \drummode {
    \omit TupletNumber
    \omit TupletBracket
    s4 s4 s4 s4
    s4 \tuplet 3/2 { s4. } s4 s4
    \set repeatCountVisibility = #(every-nth-repeat-count-visible 0)
    \repeat percent 1 {
        s4 \tuplet 3/2 { s4. } s4 s4
        \tuplet 3/2 { s4. } s4 s4 s4
    }
    s4 s4 s4 s4
    s4 s4 s4 s4
    \tuplet 3/2 { s4. } s4 s4 s4
    s4 s4 s4 s4
    s4 s4 s4 s4
}

\new DrumStaff <<
    \set DrumStaff.drumStyleTable = #(alist->hash-table style_table)
    \new DrumVoice { \voiceOne \up }
    \new DrumVoice { \voiceTwo \down }
>>
